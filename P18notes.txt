forming a SN from a string
"[1,2]" -> pair "1,2]"
    (N 1,",2]") <- sn "1,2]"
    consume ","
    (N 2,"]") <- sn "2]"
    consume "]"
(P (N 1, N 2), "")

"[[1,2],3]" -> pair "[1,2],3]"
    (P (N 1, N2),",3]") <- sn "[1,2],3]"
    consume ","
    (N 3,"]") <- sn "3]"
    consume "]"
(P (P (N 1, N 2), N 3), "")
